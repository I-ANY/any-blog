配置主机间免密ssh#

# 主机安装：openssh-server
yum -y install openssh-server
# 客户端安装：
yum -y install openssh-client
 
# 1、主机执行（连续回车），生成密钥对
ssh-keygen -t rsa
  -t 指定算法，也可指定其他算法
 
# 在~/.ssh/目录下会生成两文件：
id_rsa：私钥文件
id_rsa.pub：公钥文件
# 2、将公钥copy到其他机器上即可：
ssh-copy-id -i ~/.ssh/id_rsa.pub root@10.0.0.3
  -i : 指定公钥文件
 
# 3、删除免密：
执行上诉命令之后会在目标服务器对应用户下的：~/.ssh/authorized_keys 文件中添加一行内容，也就是生成的本地公钥
编辑该文件，删除对应主机的公钥内容，也就删除了对应主机的免密操作

openssh升级#

OpenSSH（OpenBSD Secure Shell）是OpenBSD计划组的一套用于安全访问远程计算机的连接工具。该工具是SSH协议的开源实现，支持对所有的传输进行加密，可有效阻止窃听、连接劫持以及其他网络级的攻击。 OpenSSH 8.3p1及之前版本中scp的scp.c文件存在操作系统命令注入漏洞。该漏洞源于外部输入数据构造操作系统可执行命令过程中，网络系统或产品未正确过滤其中的特殊字符、命令等。攻击者可利用该漏洞执行非法操作系统命令。 详细参考国家信息安全漏洞库信息: CVE-2020-15778 目前Rocky Linux 8.3 RC1 采用的为OpenSSH_8.0p1，所以建议通过升级OpenSSH修复此问题。同样OpenSSH before 8.5也存在漏洞，所以建议修复至最新版本OpenSSH_8.7p1。 注：此操作步骤同样适用于Red Hat系所有7.x 8.x服务器系统。

# 安装依赖
yum install wget gcc openssl-devel pam-devel rpm-build zlib -y

# 查看当前包版本
rpm -qa | grep openssh

# 备份现有的SSH
mv /etc/ssh/ /etc/ssh.bak
mv /usr/bin/ssh /usr/bin/ssh.bak
mv /usr/sbin/sshd /usr/sbin/sshd.bak
mv /etc/init.d/sshd /etc/init.d/sshd.bak   # 如果是第一次升级，备份/etc/init.d/sshd时会不存在，不影响后续操作

# 卸载当前版本
rpm -e --nodeps $(rpm -qa |grep openssh)
# 检查是否卸载完成（无结果则卸载完成）
rpm -qa | grep openssh

# 下载包
wget https://cdn.openbsd.org/pub/OpenBSD/OpenSSH/portable/openssh-8.7p1.tar.gz
# 解压
tar xvf openssh-8.7p1.tar.gz
# 进入目录
cd openssh-8.7p1
# 编译安装
./configure --prefix=/usr --sysconfdir=/etc/ssh --with-md5-passwords --with-zlib --with-ssl-dir=/usr/local/openssl --without-hardening
make && make install

chmod 600 /etc/ssh/*
cp -a contrib/redhat/sshd.init /etc/init.d/sshd && cp -a contrib/redhat/sshd.pam /etc/pam.d/sshd.pam

echo "PermitRootLogin yes" >>/etc/ssh/sshd_config

 systemctl enable sshd --now
